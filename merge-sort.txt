#include<iostream>
using namespace std;
#define MAX 9999999
#define LEN 1000

void merge(int *A, int p, int q, int r)
{
	int n1 = q - p + 1;
	int n2 = r - q;
	int L[LEN],R[LEN];
	int i, j;

	for (i = 1; i <= n1; i++)
		L[i] = A[p + i - 1];

	for (j = 1; j <= n2; j++)
		R[j] = A[q + j];//Ô­À´´íÎóÎªR[j]=A[q+i]

	L[n1 + 1] = MAX;
	R[n2 + 1] = MAX;

	i = 1, j = 1;
	int k;
	for (k = p; k <= r; k++)
	{
		if (L[i] < R[j])
		{
			A[k] = L[i];
			i++;
		}
		else
		{
			A[k] = R[j];
			j++;
		}
	}

}

void mergesort(int *A, int p, int r)
{
	if (p < r)
	{
		int m = (p + r) / 2;
		mergesort(A, p, m);
		mergesort(A, m + 1, r);
		merge(A, p, m, r);
	}
}

int main()
{
	int a[LEN], n;
	cin >> n;

	for (int i = 1; i <= n; i++)
	{
		cin >> a[i];
	}
	mergesort(a, 1, n);
	
	for (int i = 1; i <= n; i++)
	{
		cout << a[i]<<" ";
	}
	return 0;
}
